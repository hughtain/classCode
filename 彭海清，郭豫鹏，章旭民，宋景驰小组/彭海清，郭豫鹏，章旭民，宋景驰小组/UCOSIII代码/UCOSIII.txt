	while(1)
	{
			  
   pmsg=OSQPend(&HUM_queue,0,OS_OPT_PEND_NON_BLOCKING,&msg_size,0,&err);
      
		if(pmsg=="ON")
		 {
			 
			 
			 machine1_goto();
			 
			
			 
		 }

 		 {
			 
				 
			 machine1_back();
			
			 }

 		 {
			 
				 
			 machine1_fast_stop();
			
			 }
		 			 
		
		 
		OSTimeDlyHMSM(0,0,0,1,OS_OPT_TIME_DLY,&err); //??200ms

	}
	
	
}

void WIN_task(void *p_arg)
{
	OS_ERR err;
	CPU_SR_ALLOC();
	char *pmsg;
	int flag=0,flag1=0;
	OS_MSG_SIZE msg_size;
	p_arg = p_arg;
	
	while(1)
	{
			  
   pmsg=OSQPend(&WIN_queue,0,OS_OPT_PEND_NON_BLOCKING,&msg_size,0,&err);
       OSSchedLock(&err);
		if(pmsg=="ON")
		 {
			 if(flag==0)
			 {
				 flag=1;
				 flag1=0;
			 
			 machine_goto();
			 unitdelay_ms(500);
			 machine_fast_stop();
			 }
			 
		 }
		 if(pmsg=="OFF")
 		 {
			 if(flag1==0)
			 {
				 flag1=1;
				 flag=0;
				 
			 machine_back();
			 unitdelay_ms(500);
			 machine_fast_stop();
			 }
		 }			 
		OSSchedUnlock(&err);
		 pmsg=NULL;
		OSTimeDlyHMSM(0,0,0,1,OS_OPT_TIME_DLY,&err); //??200ms

	}
	
	
}
